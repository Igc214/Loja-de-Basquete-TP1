' --- CLASSES PRINCIPAIS ---

' Mudamos para abstract: não pode ser instanciada sozinha
class TenisPerformance {
    - id: int
    - nome: String
    - numeroDoPe: int
    - cor: String
    - descricao: String
    - edicaoLimitada: int
    - autografado: boolean
}

class Review {
 - url: String
}
   
    

abstract class Pessoa {
    - id: int
    - apelido: String
    - telefone: String
    - endereco: String
    - email: String
}

' --- HERANÇA (ESPECIALIZAÇÕES) ---

' Agora TenisPerformance É UM Tenis (Extends)


class Cliente extends Pessoa {
    - cpf: String
    - numeroDeCompra: int
}

class Funcionario extends Pessoa {
    - cpf: String
    - cargo: String
    - salario: double
}

class Fornecedor extends Pessoa {
    - cnpj: String
}

' --- COMPOSIÇÃO E ASSOCIAÇÕES ---

class Fabricante {
    - cnpj: String
    - nomeMarca: String
}

enum Posicao {
    +PG
    +SG
    +SF
    +PF
    + C
}

enum TipoSolado {
    +INDOOR 
    +OUTDOOR 
    +HIBRIDO
}

' --- RELACIONAMENTOS ---

' Relacionamentos da classe base Tenis
TenisPerformance "*" --> "1" Posicao
TenisPerformance "*" --> "1" TipoSolado
TenisPerformance "*" o--> "1" Fabricante
TenisPerformance "1" *--> "*" Review

' Relacionamento Fabricante x Fornecedor
Fabricante "*" o--> "*" Fornecedor 

@enduml
